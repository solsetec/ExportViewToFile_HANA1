CREATE PROCEDURE SOL_SP_EXPORTVIEW(ETAPA NVARCHAR(100)
	,MENSAJE NVARCHAR(1000)
)
LANGUAGE SQLSCRIPT AS


BEGIN
INSERT INTO "@SOL_LOGS"
			(
			"Code"
			,"Name"
			,"U_SOL_Etapa" 
			,"U_SOL_Mensaje"
			)
		VALUES((SELECT LPAD(TO_NVARCHAR(IFNULL(MAX(TO_INT("Code")), 0) + 1), 2, '0') FROM "@SOL_LOGS"),
			  	(SELECT CURRENT_TIMESTAMP FROM DUMMY)
			  	,:ETAPA
			  	,:MENSAJE);
		
END
